Anlisis del Problema de Monty Hall

Un concursante debe elegir 1 puerta entre 3 (todas cerradas); el premio consiste en llevarse lo que se encuentra detrs de la puerta elegida. Se sabe con certeza que tras una de ellas hay un premio, y tras las otras dos no hay premio.

Una vez que el concursante haya elegido una puerta y comunicado su eleccin a los presentes, el presentador, que sabe lo que hay detrs de cada puerta, abrir una de las otras dos, en la que no habr premio. A continuacin, le da la opcin al concursante de cambiar, si lo desea, de puerta.

Pregunta 1. Debe el concursante mantener su eleccin original o escoger la otra puerta?

Pregunta 2. Existe alguna diferencia en la estrategia si en lugar de 3 se tienen 5 puertas (1 con premio y 4 sin premio)?

puertas = [0, 1, 2]
ganar_quedarse = Fraction(0, 1)
ganar_cambio= Fraction(0, 1)
total_prob = Fraction(0, 1)

for ganar in puertas:
    for seleccion in puertas:

        prob_inicial = Fraction(1, 9)

        aperturas_posibles = [d for d in puertas if d != ganar and d != seleccion]

        for abierta in aperturas_posibles:
            host_prob = Fraction(1, len(aperturas_posibles))
            scenario_prob = prob_inicial * host_prob

            if seleccion == ganar:
                ganar_quedarse += scenario_prob

            switch_seleccion = next(d for d in puertas if d not in (seleccion, abierta))
            if switch_seleccion == ganar:
                ganar_cambio+= scenario_prob

            total_prob += scenario_prob

print("Probabilidad quedarse:  ", float(ganar_quedarse), "=", ganar_quedarse)
print("Probabilidad cambiar:    ", float(ganar_cambio), "=", ganar_cambio)

from fractions import Fraction

puetas5 = list(range(5))
quedarse_ganar = Fraction(0, 1)
cambiar_ganar = Fraction(0, 1)
total_prob = Fraction(0, 1)

for ganar5 in puetas5:
    for choice in puetas5:

        initial_prob = Fraction(1, 25)

        # Si el jugador se queda con su puerta y esa es la ganadora
        if choice == ganar5:
            quedarse_ganar += initial_prob

        # El anfitrin abre 3 puertas distintas que no son la ganadora ni la elegida
        possible_opens = [d for d in puetas5 if d != ganar5 and d != choice]

        if len(possible_opens) >= 3:
            num_host_options = len(possible_opens) * (len(possible_opens) - 1) * (len(possible_opens) - 2) // 6
            host_prob = Fraction(1, num_host_options)
            scenario_prob = initial_prob * host_prob

            remaining = [d for d in puetas5 if d != choice and d not in possible_opens]
            if len(remaining) == 1 and remaining[0] == ganar5:
                cambiar_ganar += scenario_prob

            total_prob += scenario_prob

print("Probabilidad terica de ganar quedndote: ", float(quedarse_ganar), "=", quedarse_ganar)
print("Probabilidad terica de ganar cambiando:   ", float(cambiar_ganar), "=", cambiar_ganar)

Probabilidad terica de ganar cambiando:    0.8 = 4/5

import random
import csv

def puerta_elegida(n):
    if n <= 0.33:
        return 0
    elif 0.33 < n <= 0.66:
        return 1
    else:
        return 2

def puerta_elegida_5(n):
    if n <= 0.2:
        return 0
    elif 0.2 < n <= 0.4:
        return 1
    elif 0.4 < n <= 0.6:
        return 2
    elif 0.6 < n <= 0.8:
        return 3
    else:
        return 4

def simulacion_3_puertas_sin_cambiar():
    random.seed(2025)
    registros = []
    
    for i in range(10**5):
        # Se pone el premio
        p = random.random()
        premio = puerta_elegida(p)
        
        # El usuario elige una puerta
        n = random.random()
        eleccion = puerta_elegida(n)
        
        # El presentador abre una puerta
        e = random.random()
        while (puerta_elegida(e) == eleccion) or (puerta_elegida(e) == premio):
            e = random.random()
        puerta_abierta = puerta_elegida(e)
        
        # Resultado (no cambia)
        resultado = 1 if eleccion == premio else 0
        
        registros.append({
            'simulacion_id': i,
            'num_puertas': 3,
            'premio': premio,
            'eleccion_inicial': eleccion,
            'puerta_abierta': puerta_abierta,
            'estrategia': 'No cambiar',
            'resultado': resultado,
        })
    
    return registros

def simulacion_3_puertas_con_cambio():
    random.seed(2025)
    registros = []
    puertas_posibles = [0, 1, 2]
    
    for i in range(10**5):
      
        p = random.random()
        premio = puerta_elegida(p)
        
   
        n = random.random()
        eleccion = puerta_elegida(n)
        
        e = random.random()
        while (puerta_elegida(e) == eleccion) or (puerta_elegida(e) == premio):
            e = random.random()
        puerta_abierta = puerta_elegida(e)
        
        puertas_restantes = [p for p in puertas_posibles if p != eleccion and p != puerta_abierta]
        nueva_eleccion = puertas_restantes[0]

            'resultado': resultado,

    return registros

def simulacion_5_puertas_con_cambio():
    random.seed(2025)
    registros = []
    posibilidades = [0, 1, 2, 3, 4]

    for i in range(10**5):
        p = random.random()
        premio = puerta_elegida_5(p)

        n = random.random()
        eleccion_inicial = puerta_elegida_5(n)

        # El anfitrin abre 3 puertas que no son la elegida ni la ganadora
        opciones_para_abrir = list(set(posibilidades) - set([eleccion_inicial, premio]))
        puertas_abiertas = random.sample(opciones_para_abrir, 3)

        # El jugador cambia a una de las puertas restantes (solo queda una)
        puertas_no_disponibles = set(puertas_abiertas + [eleccion_inicial])
        opciones_finales = list(set(posibilidades) - puertas_no_disponibles)
        final_choise = opciones_finales[0]

        resultado = 1 if final_choise == premio else 0

        registros.append({
            'simulacion_id': i,
            'num_puertas': 5,
            'premio': premio,
            'eleccion_inicial': eleccion_inicial,
            'puertas_abiertas': puertas_abiertas,
            'eleccion_final': final_choise,
            'estrategia': 'Cambiar',
            'resultado': resultado,
        })

    return registros




def guardar_csv(datos, nombre_archivo):
    with open(nombre_archivo, 'w', newline='') as csvfile:
        campos = datos[0].keys()
        writer = csv.DictWriter(csvfile, fieldnames=campos)
        
        writer.writeheader()
        for registro in datos:
            writer.writerow(registro)
